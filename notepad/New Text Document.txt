--------------While loop-------------
const fruits = ["Banana", "Apple", "Jackfruit", "Pinapple", "Orange"]

i=0;
while( i<fruits.length ){
    console.log(fruits[i]);
    i++;
}



----------for loop-----------
const fruits = ["Banana", "Apple", "Jackfruit", "Pinapple", "Orange"]

for( i=0; i<fruits.length; i++ ){
    console.log(fruits[i]);
}
